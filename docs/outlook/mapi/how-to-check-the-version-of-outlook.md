    HRESULT hr = E_FAIL;
    LPTSTR pszTempPath = NULL;
    LPTSTR pszTempVer = NULL;

            pszOutlookQualifiedComponents[i],

    {
        for (i = 0; i < nOutlookQualifiedComponents; i++)

                pszOutlookQualifiedComponents[i],
                TEXT("outlook.exe"),
                (DWORD) INSTALLMODE_DEFAULT,
                NULL,
                &dwValueBuf);

    {

                TEXT("outlook.exe"),
                (DWORD) INSTALLMODE_EXISTING,
